% funciones 
function [L,U] = lu1(A)
% Eliminacion Gaussiana sin pivoteo
m = size(A,1); % m = size(A,2);
U = A; L = eye(m); 
for k = 1:m-1
    for j = k+1:m
        L(j,k) = U(j,k)/U(k,k);
        U(j,k:m) = U(j,k:m)-L(j,k)*U(k,k:m);
    end
end
end

function [L,U,P] = lu2(A)
% Eliminacion Gaussiana con pivoteo
m = size(A,1); % m = size(A,2);
U = A; L = eye(m); P = eye(m);
for k = 1:m-1
    % intercambio filas
    [~,i] = max(abs(U(k:m,k)));
    i = i + k - 1;
    % intercambio columna k, i 
    %P(:,k) % columna k
    %P(:,i) % columna i
    P([k i],:) = P([i k],:);
    % U(k,k:m), U(i,k:m)
    U([k,i],k:m)   = U([i,k],k:m);
    L([k,i],1:k-1) = L([i,k],1:k-1);
    %
    for j = k+1:m
        L(j,k) = U(j,k)/U(k,k);
        U(j,k:m) = U(j,k:m)-L(j,k)*U(k,k:m);
    end
end
end
